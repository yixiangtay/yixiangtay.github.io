This post contains code snippets for recursive/iterative versions of breadth/depth-first search.

Breadth-First Search

  1. Level Order Traversal
import collections

def recursive_level_order(queue, res):
    if not queue:
        return

    node = queue.popleft()
    res.append(node.val) # process node
    if node.left:
        queue.append(node.left)
    if node.right:
        queue.append(node.right)

    return recursive_level_order(queue, res)

def iterative_level_order(root):
    queue = collections.deque([root])
    res = []

    while queue:
        node = queue.popleft()
        res.append(node.val) # process node
        if node.left:
            queue.append(node.left)
        if node.right:
            queue.append(node.right)

    return res

Depth-First Search

  1. Inorder Traversal
def recursive_inorder(root, res):
    if root:
        recursive_inorder(root.left, res)
        res.append(root.val) # process node
        recursive_inorder(root.right, res)

def iterative_inorder(root):
    stack = []
    res = []

    while stack or root:
        while root:
            stack.append(root)
            root = root.left
        root = stack.pop()
        res.append(root.val) # process node
        root = root.right

    return res
  1. Preorder Traversal
def recursive_preorder(root, res):
    if root:
        res.append(root.val) # process node
        recursive_preorder(root.left, res)
        recursive_preorder(root.right, res)

def iterative_preorder(root):
    stack = []
    res = []

    while stack or root:
        while root:
            res.append(root.val) # process node
            stack.append(root)
            root = root.left
        root = stack.pop()
        root = root.right

    return res
  1. Postorder Traversal
def recursive_postorder(root, res):
    if root:
        recursive_postorder(root.left, res)
        recursive_postorder(root.right, res)
        res.append(root.val) # process node

def iterative_postorder(root):
    stack = []
    res = [] 

    while stack or root:
        while root:
            res.append(root.val) # process node
            stack.append(root.left)
            root = root.right
        root = stack.pop()

    res.reverse()
    return res
